## Install

```bash
npm install --save react-switch-tis
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-switch-tis'
import 'react-switch-tis/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
